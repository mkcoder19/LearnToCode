Python Control Flow

Conditional Statements

If-Else

age = 18
if age >= 18:
    print("You are an adult")
else:
    print("You are a minor")

Elif

score = 85
if score >= 90:
    grade = 'A'
elif score >= 80:
    grade = 'B'
elif score >= 70:
    grade = 'C'
else:
    grade = 'D'

Loops

For Loop

# Iterate through a list
fruits = ['apple', 'banana', 'cherry']
for fruit in fruits:
    print(fruit)

# Using range
for i in range(5):  # 0 to 4
    print(i)

While Loop

count = 0
while count < 5:
    print(count)
    count += 1
