C++ Object-Oriented Programming

Classes and Objects

Classes are user-defined types that encapsulate data and functions.

class Rectangle {
private:
    double width;
    double height;
    
public:
    // Constructor
    Rectangle(double w, double h) : width(w), height(h) {}
    
    // Member function
    double area() {
        return width * height;
    }
    
    // Setter
    void setWidth(double w) {
        width = w;
    }
};

// Creating objects
Rectangle rect(5.0, 3.0);
cout << "Area: " << rect.area(); // 15.0

Inheritance

Derived classes inherit properties and behaviors from base classes.

class Shape {
protected:
    string color;
    
public:
    Shape(string c) : color(c) {}
    virtual double area() = 0; // Pure virtual function
};

class Circle : public Shape {
    double radius;
    
public:
    Circle(string c, double r) : Shape(c), radius(r) {}
    
    double area() override {
        return 3.14159 * radius * radius;
    }
};

Circle circle("Red", 5.0);
cout << "Area: " << circle.area();

Polymorphism

Objects can take many forms through inheritance and virtual functions.

Shape* shapes[2];
shapes[0] = new Circle("Red", 5.0);
shapes[1] = new Rectangle("Blue", 4.0, 6.0);

for (int i = 0; i < 2; i++) {
    cout << "Area: " << shapes[i]->area() << endl;
}

Encapsulation

Bundling data with methods that operate on that data.

class BankAccount {
private:
    double balance;
    
public:
    void deposit(double amount) {
        if (amount > 0) balance += amount;
    }
    
    double getBalance() {
        return balance;
    }
};

BankAccount account;
account.deposit(1000);
cout << "Balance: " << account.getBalance();
