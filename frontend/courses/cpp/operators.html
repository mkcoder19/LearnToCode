C++ Operators

Arithmetic Operators

Operator Description Example
+ Addition int sum = 5 + 3; // 8
- Subtraction int diff = 5 - 3; // 2
* Multiplication int product = 5 * 3; // 15
/ Division int quotient = 5 / 2; // 2
% Modulus int remainder = 5 % 2; // 1
++ Increment int x = 5; x++; // 6
-- Decrement int y = 5; y--; // 4

Practical Examples

// Calculate circle area
double radius = 5.0;
double area = 3.14159 * radius * radius;

// Integer vs floating-point division
int int_div = 5 / 2;    // 2
double float_div = 5.0 / 2; // 2.5

Relational and Logical Operators

Operator Description Example
== Equal to 5 == 3 // false
!= Not equal 5 != 3 // true
> Greater than 5 > 3 // true
< Less than 5 < 3 // false
&& Logical AND true && false // false
|| Logical OR true || false // true
! Logical NOT !true // false

Bitwise Operators

// Bitwise AND
int a = 5;    // 0101
int b = 3;    // 0011
int result = a & b; // 0001 (1)

// Bitwise OR
result = a | b; // 0111 (7)

// Bitwise XOR
result = a ^ b; // 0110 (6)

// Bitwise NOT
result = ~a; // 11111010 (-6)

// Left shift
result = a << 1; // 1010 (10)

// Right shift
result = a >> 1; // 0010 (2)
