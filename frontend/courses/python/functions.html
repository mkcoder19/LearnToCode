Python Functions

Defining Functions

# Basic function
def greet():
    print("Hello, World!")

# Function with parameters
def greet_name(name):
    print(f"Hello, {name}!")

# Function with return value
def add_numbers(a, b):
    return a + b

Function Arguments

Default Arguments

def greet(name="Guest"):
    print(f"Hello, {name}!")

greet()        # Hello, Guest!
greet("Alice") # Hello, Alice!

Keyword Arguments

def describe_pet(pet_name, animal_type="dog"):
    print(f"I have a {animal_type} named {pet_name}")

describe_pet(animal_type="hamster", pet_name="Harry")
